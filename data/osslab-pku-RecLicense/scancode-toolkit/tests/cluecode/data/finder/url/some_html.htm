>>> import re
>>> URLS_REGEX=re.compile('''((?:(?:ht|f)tp[s]?://(?:[0-9a-zA-Z][-\w]*[0-9a-zA-Z]\.)+[a-zA-Z]{2,9})(?::\d{1,4})?[-\w\/#~:.?+=&%@~]*[-\w\/#~:?+=&%@~]{1})''')
